import { DateTime } from 'luxon';

export const MARKET_TIMEZONE = 'America/New_York';

export function getMarketTime(): DateTime {
  return DateTime.now().setZone(MARKET_TIMEZONE);
}

export function getMarketTimestamp(): number {
  return getMarketTime().toMillis();
}

export function formatMarketTime(format: string = 'yyyy-MM-dd HH:mm:ss'): string {
  return getMarketTime().toFormat(format);
}

export function isMarketOpen(): boolean {
  const now = getMarketTime();
  const hour = now.hour;
  const minute = now.minute;
  const day = now.weekday;
  
  if (day === 6 || day === 7) return false;
  
  const marketOpen = 9 * 60 + 30;
  const marketClose = 16 * 60;
  const currentTime = hour * 60 + minute;
  
  return currentTime >= marketOpen && currentTime < marketClose;
}

export function getMarketSession(): 'pre-market' | 'market' | 'after-hours' | 'closed' {
  const now = getMarketTime();
  const hour = now.hour;
  const minute = now.minute;
  const day = now.weekday;
  
  if (day === 6 || day === 7) return 'closed';
  
  const currentTime = hour * 60 + minute;
  
  if (currentTime < 9 * 60 + 30) return 'pre-market';
  if (currentTime < 16 * 60) return 'market';
  if (currentTime < 20 * 60) return 'after-hours';
  return 'closed';
}

export function getNextMarketOpen(): DateTime {
  const now = getMarketTime();
  let nextOpen = now.set({ hour: 9, minute: 30, second: 0, millisecond: 0 });
  
  if (now.weekday === 6) {
    nextOpen = nextOpen.plus({ days: 2 });
  } else if (now.weekday === 7) {
    nextOpen = nextOpen.plus({ days: 1 });
  } else if (!isMarketOpen() && now.hour >= 16) {
    nextOpen = nextOpen.plus({ days: 1 });
    if (nextOpen.weekday === 6) nextOpen = nextOpen.plus({ days: 2 });
  }
  
  return nextOpen;
}
